## Task 4: Create i18n Routing Config

**Files:**
- Create: `i18n/routing.ts`

- [ ] **Step 1: Create i18n directory and routing.ts**

Create `i18n/routing.ts`:

```ts
import { defineRouting } from 'next-intl/routing'

export const routing = defineRouting({
  locales: ['de', 'fr', 'en', 'it'],
  defaultLocale: 'de',
})

export type Locale = (typeof routing.locales)[number]
```

- [ ] **Step 2: Create i18n/request.ts**

Create `i18n/request.ts`:

```ts
import { getRequestConfig } from 'next-intl/server'
import { routing } from './routing'

export default getRequestConfig(async ({ requestLocale }) => {
  let locale = await requestLocale

  // Validate that the incoming locale is supported
  if (!locale || !routing.locales.includes(locale as (typeof routing.locales)[number])) {
    locale = routing.defaultLocale
  }

  return {
    locale,
    messages: (await import(`../messages/${locale}.json`)).default,
  }
})
```

```bash

git commit -m "feat: add next-intl routing config and request config"
```

```ts
// proxy.ts — Next.js 16 proxy file (ehemals middleware.ts)
import { getToken } from 'next-auth/jwt'
import { NextResponse } from 'next/server'
import type { NextRequest } from 'next/server'
import createIntlProxy from 'next-intl/proxy'
import { routing } from './i18n/routing'

const intlProxy = createIntlProxy(routing)

export async function proxy(req: NextRequest) {
  // 1. next-intl locale routing (läuft zuerst)
  const intlResponse = intlProxy(req)
  if (intlResponse) return intlResponse

  const { pathname } = req.nextUrl

  // 2. Öffentliche Routen immer durchlassen
  // Locale-Präfix aus dem Pfad entfernen für Vergleich
  const pathnameWithoutLocale = pathname.replace(/^\/(de|fr|en|it)/, '') || '/'

  if (
    pathnameWithoutLocale.startsWith('/auth') ||
    pathnameWithoutLocale.startsWith('/403')
  ) {
    return NextResponse.next()
  }

  const token = await getToken({ req, secret: process.env.NEXTAUTH_SECRET })

  // 3. Nicht eingeloggt → Login (mit Locale)
  if (!token) {
    const locale = pathname.split('/')[1] || 'de'
    return NextResponse.redirect(new URL(`/${locale}/auth/login`, req.url))
  }

  const role = token.role as string

  if (pathnameWithoutLocale.startsWith('/dashboard') && role === 'MIETER') {
    return NextResponse.redirect(new URL('/403', req.url))
  }

  if (pathnameWithoutLocale.startsWith('/tenant') && role !== 'MIETER') {
    return NextResponse.redirect(new URL('/403', req.url))
  }

  if (pathnameWithoutLocale.startsWith('/superadmin') && role !== 'SUPER_ADMIN') {
    return NextResponse.redirect(new URL('/403', req.url))
  }

  if (
    pathnameWithoutLocale.startsWith('/dashboard/team') &&
    role !== 'ADMIN' &&
    role !== 'SUPER_ADMIN'
  ) {
    return NextResponse.redirect(new URL('/403', req.url))
  }

  return NextResponse.next()
}

export const config = {
  matcher: [
    // Alle Routen ausser _next, static files, api routes
    '/((?!_next|api|uploads|.*\\..*).*)',
  ],
}
```

**Note:** `createIntlProxy` is the next-intl proxy helper — verify exact import name in next-intl docs after installation. If the API differs, check `node_modules/next-intl/proxy/index.js` for the correct export.
